Для создания простого веб-расписания с днями недели, временем с 10:00 до 22:00 и возможностью внесения изменений и сохранения данных, можно использовать HTML и JavaScript. Для сохранения данных между сессиями можно использовать localStorage, так как это позволяет сохранить данные прямо в браузере без серверной стороны. Я также добавлю список учеников, которых можно будет добавлять.

 

Пример кода:

 

<!DOCTYPE html>

<html lang="ru">

<head>

    <meta charset="UTF-8">

    <meta name="viewport" content="width=device-width, initial-scale=1.0">

    <title>Расписание уроков</title>

    <style>

        body {

            font-family: Arial, sans-serif;

            background-color: #f4f4f4;

            margin: 20px;

        }

        table {

            width: 100%;

            border-collapse: collapse;

        }

        th, td {

            border: 1px solid #ddd;

            padding: 8px;

            text-align: center;

        }

        th {

            background-color: #f2f2f2;

        }

        input, select {

            width: 100%;

            padding: 5px;

        }

        .actions {

            margin-top: 20px;

        }

        .input-container {

            margin-bottom: 10px;

        }

    </style>

</head>

<body>

 

<h1>Расписание уроков</h1>

 

<!-- Список учеников -->

<div class="input-container">

    <label for="studentName">Добавить ученика:</label>

    <input type="text" id="studentName" placeholder="Имя ученика">

    <button onclick="addStudent()">Добавить ученика</button>

</div>

 

<h3>Список учеников:</h3>

<ul id="studentList">

    <!-- Список учеников будет отображаться здесь -->

</ul>

 

<!-- Расписание -->

<h3>Расписание</h3>

<table>

    <thead>

        <tr>

            <th>Время</th>

            <th>Понедельник</th>

            <th>Вторник</th>

            <th>Среда</th>

            <th>Четверг</th>

            <th>Пятница</th>

            <th>Суббота</th>

            <th>Воскресенье</th>

        </tr>

    </thead>

    <tbody id="schedule">

        <!-- Строки расписания будут сгенерированы через JavaScript -->

    </tbody>

</table>

 

<div class="actions">

    <button onclick="saveSchedule()">Сохранить изменения</button>

    <button onclick="loadSchedule()">Загрузить расписание</button>

</div>

 

<script>

    // Список учеников

    let students = JSON.parse(localStorage.getItem('students')) || [];

 

    // Функция для добавления ученика

    function addStudent() {

        const studentName = document.getElementById('studentName').value.trim();

        if (studentName) {

            students.push(studentName);

            document.getElementById('studentName').value = ''; // Очистить поле ввода

            updateStudentList();

            saveStudents();

        }

    }

 

    // Функция для обновления списка учеников

    function updateStudentList() {

        const studentListElement = document.getElementById('studentList');

        studentListElement.innerHTML = '';

        students.forEach(student => {

            const li = document.createElement('li');

            li.textContent = student;

            studentListElement.appendChild(li);

        });

    }

 

    // Сохранение списка учеников в localStorage

    function saveStudents() {

        localStorage.setItem('students', JSON.stringify(students));

    }

 

    // Генерация расписания

    const times = [];

    for (let i = 10; i <= 22; i++) {

        times.push(i + ':00');

    }

 

    function generateSchedule() {

        const scheduleTable = document.getElementById('schedule');

        scheduleTable.innerHTML = ''; // Очистить таблицу перед рендером

 

        times.forEach(time => {

            const row = document.createElement('tr');

            const timeCell = document.createElement('td');

            timeCell.textContent = time;

            row.appendChild(timeCell);

 

            for (let i = 0; i < 7; i++) {

                const cell = document.createElement('td');

                const input = document.createElement('input');

                input.type = 'text';

                input.id = `schedule-${time}-${i}`;

                cell.appendChild(input);

                row.appendChild(cell);

            }

 

            scheduleTable.appendChild(row);

        });

    }

 

    // Сохранение расписания в localStorage

    function saveSchedule() {

        const scheduleData = [];

        times.forEach(time => {

            const rowData = [];

            for (let i = 0; i < 7; i++) {

                const input = document.getElementById(`schedule-${time}-${i}`);

                rowData.push(input.value);

            }

            scheduleData.push({ time: time, data: rowData });

        });

 

        localStorage.setItem('schedule', JSON.stringify(scheduleData));

    }

 

    // Загрузка расписания из localStorage

    function loadSchedule() {

        const scheduleData = JSON.parse(localStorage.getItem('schedule')) || [];

        scheduleData.forEach(item => {

            const time = item.time;

            const rowData = item.data;

            rowData.forEach((value, index) => {

                const input = document.getElementById(`schedule-${time}-${index}`);

                input.value = value;

            });

        });

    }

 

    // Инициализация страницы

    generateSchedule();

    updateStudentList();

</script>

 

</body>

</html>
